fsWriteText.mjs

import fs from 'fs'


/**
 * 寫入utf-8文字至檔案
 *
 * @export
 * @param {String} fn 輸入檔案名稱
 * @param {String} c 輸入utf-8文字數據
 */
export default function fsWriteText(fn, c) {
    fs.writeFileSync(fn, c, 'utf8')
}
