    //check
    if (!isarr(matOld)) {
        return {}
    }
    if (!isarr(matNew)) {
        return {}
    }
    if (size(matOld) === 0 && size(matNew) === 0) {
        return {}
    }

    //arrDiff
    let r = arrDiff(matOld, matNew, { ...opt, mode: 'matrix' })

    return r
}
